Funkcja Opis Przykład
fopen Otwiera plik i zwraca "uchwyt" do niego.
Trzy podstawowe tryby otwierania pliku:
      w - do zapisu
      r - do odczyty
      a - do dopisywania
 
$plik = fopen("lista.txt","w");
$dane = fopen("dane/osoby/pawel.txt","r");
$konto = fopen("../bank/konto.txt","a");
fgets Wczytuje z pliku pojedynczą linię. 
$plik = fopen("dane/osoby/pawel.txt","r");
$pierwszaLinia = fgets($plik);
$drugaLinia = fgets($plik);
fclose Zamyka plik 
$plik = fopen("dane/osoby/pawel.txt","r");
fclose($plik);
eof
